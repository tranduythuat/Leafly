function switchTheme(theme) {
  currentTheme.value = theme
  const heroSection = sections.value.find(s => s.type === 'hero')
  if (!heroSection) return
  const themes = {
    garden:  'linear-gradient(135deg,#eaede4 0%,#f4f0e3 50%,#fbe8e4 100%)',
    blush:   'linear-gradient(135deg,#fbe8e4 0%,#fdf5f3 50%,#f8d5cd 100%)',
    classic: 'linear-gradient(135deg,#f3f1ec 0%,#faf9f7 50%,#e8e4db 100%)',
  }
  // Replace background in hero html
  heroSection.html = heroSection.html.replace(/background:[^;]+;/, `background:${themes[theme]};`)
  showToast(`${theme.charAt(0).toUpperCase() + theme.slice(1)} theme applied 🎨`)
}

// ==================== UNDO / REDO ====================
function saveState() {
  const snapshot = JSON.stringify(sections.value)
  undoStack.value.push(snapshot)
  if (undoStack.value.length > 30) undoStack.value.shift()
  redoStack.value = []
}

function undo() {
  if (undoStack.value.length < 2) return
  const current = undoStack.value.pop()
  redoStack.value.push(current)
  sections.value = JSON.parse(undoStack.value[undoStack.value.length - 1])
  showToast('Undo ↩')
}

function redo() {
  if (!redoStack.value.length) return
  const state = redoStack.value.pop()
  undoStack.value.push(state)
  sections.value = JSON.parse(state)
  showToast('Redo ↪')
}

    saveStatus.value = 'saved'

    showToast('Invitation saved! 💾')
    setTimeout(() => {
      saveStatus.value = 'pulse'
      saveLabel.value = 'Auto-saved'
    }, 2000)
  }, 800)
}

  const id = Date.now()
  toasts.value.push({ id, msg })
  setTimeout(() => {
    toasts.value = toasts.value.filter(t => t.id !== id)
  }, 2500)
}
