<template>
  <aside class="editor-sidebar">
    <section class="editor-sidebar__group">
      <div class="editor-sidebar__group-header">
        <span>Theme Presets</span>
        <span class="editor-sidebar__group-meta">2</span>
      </div>

      <div class="editor-sidebar__theme-grid">
        <button class="editor-sidebar__theme-card editor-sidebar__theme-card--sage" type="button">
          Sage
        </button>
        <button class="editor-sidebar__theme-card editor-sidebar__theme-card--blush" type="button">
          Blush
        </button>
      </div>
    </section>

    <section class="editor-sidebar__group">
      <div class="editor-sidebar__group-header">
        <span>Blocks</span>
      </div>

      <div class="editor-sidebar__block-list">
        <button class="editor-sidebar__block-card" type="button" @click="insertText('heading')">
          <strong>Heading</strong>
          <span>Large title block</span>
        </button>
        <button class="editor-sidebar__block-card" type="button" @click="insertText('paragraph')">
          <strong>Paragraph</strong>
          <span>Body content block</span>
        </button>
        <button class="editor-sidebar__block-card" type="button" @click="insertText('quote')">
          <strong>Quote</strong>
          <span>Editorial pull quote</span>
        </button>
        <button class="editor-sidebar__block-card" type="button" @click="store.insertImageBlock()">
          <strong>Image</strong>
          <span>Photo placeholder</span>
        </button>
        <button class="editor-sidebar__block-card" type="button" @click="store.addSection()">
          <strong>Blank Section</strong>
          <span>Custom layout</span>
        </button>
      </div>
    </section>

    <section class="editor-sidebar__group">
      <div class="editor-sidebar__group-header">
        <span>Assets</span>
      </div>

      <div class="editor-sidebar__asset-actions">
        <label class="editor-sidebar__upload-btn">
          Upload image
          <input type="file" accept="image/*" @change="onUpload" />
        </label>
        <label class="editor-sidebar__upload-btn editor-sidebar__upload-btn--muted">
          Canvas background
          <input type="file" accept="image/*" @change="onUploadBg" />
        </label>
      </div>
    </section>

    <section class="editor-sidebar__order">
      <div class="editor-sidebar__group-header">
        <span>Section Order</span>
        <span class="editor-sidebar__group-meta">{{ store.sections.length }}</span>
      </div>

      <draggable v-model="sections" item-key="id" class="editor-sidebar__sections">
        <template #item="{ element }">
          <button
            class="editor-sidebar__section-item"
            :class="{ 'editor-sidebar__section-item--active': store.activeSectionId === element.id }"
            type="button"
            @click="store.selectSection(element.id)"
          >
            <span class="editor-sidebar__section-grip">::</span>
            <span class="editor-sidebar__section-name">{{ element.name }}</span>
          </button>
        </template>
      </draggable>
    </section>
  </aside>
</template>

<script setup lang="ts">
import { computed } from "vue";
import draggable from "vuedraggable";
import { useEditorStore } from "../../store/editorStore";

const store = useEditorStore();
const sections = computed({
  get: () => store.sections,
  set: (value) => store.replaceSections(value),
});

const insertText = (variant: "heading" | "paragraph" | "quote") => {
  store.insertTextBlock(variant);
};

const onUpload = (e: Event) => {
  const input = e.target as HTMLInputElement;
  const file = input.files?.[0];
  if (!file) return;

  const url = URL.createObjectURL(file);
  store.addImage(url);
  input.value = "";
};

const onUploadBg = (e: Event) => {
  const input = e.target as HTMLInputElement;
  const file = input.files?.[0];
  if (!file) return;

  const url = URL.createObjectURL(file);
  store.setBackground("image", url);
  input.value = "";
};
</script>
